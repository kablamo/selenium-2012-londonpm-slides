package MySelenium;
  use Moose;
  extends 'Test::WWW::Selenium::More';
  
  has host        => (...);
  has port        => (...);
  has browser     => (...);
  has browser_url => (
      is      => 'rw', 
      isa     => 'Str', 
      default => 'http://localhost:5000'
  );
  
  sub login_ok {
      my ($self, $username, $password) = @_;
      $self->open_ok('/login.tt');
      $self->type_ok(username => $username);
      $self->type_ok(password => $password);
      $self->click_ok('login');
  }
  
  no Moose;
